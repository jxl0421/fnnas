# CumeBox2启动配置文件
# T95标准启动地址配置
# 使用p212通用DTB（T95兼容）

# 启动参数
bootargs=root=/dev/mmcblk0p1 rootfstype=ext4 rootwait console=ttyS0,115200n8 console=tty0

# 启动地址 - T95标准启动地址
kernel_addr_r=0x10800000
fdt_addr_r=0x10000000
ramdisk_addr_r=0x11000000

# 设备树 - 使用T95通用DTB
fdt_file=meson-gxl-s905x-p212.dtb

# 启动镜像
boot_image=uImage
initrd_image=uInitrd
